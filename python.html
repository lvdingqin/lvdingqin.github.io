def ask_ok(prompt, retries=4, reminder='Please try again!'):
    while True:
        ok = input(prompt)
        if ok in ('y', 'ye', 'yes'):
            return True
        if ok in ('n', 'no', 'nop', 'nope'):
            return False
        retries = retries - 1
        if retries < 0:
            raise ValueError('invalid user response')
        print(reminder)

# *arguments 接收正常参数后， 关键字参数前的位置参数作为元组，必须在正常以后声明，关键字参数之前
# **keywords 接收位置参数以后的所有关键字参数作为字典，必须在正常参数以后声明
def cheeseshop(kind, *arguments, **keywords):
    print("-- Do you have any", kind, "?")
    print("-- I'm sorry, we're all out of", kind)
    for arg in arguments:
        print(arg)
    print("-" * 40)
    for kw in keywords:
        print(kw, ":", keywords[kw])

class Reverse: #使类实例支付for循环，实现__iter__, __next__方法
    """Iterator for looping over a sequence backwards."""
    def __init__(self, data):
        self.data = data
        self.index = len(data)

    def __iter__(self):
        return self

    def __next__(self):
        if self.index == 0:
            raise StopIteration
        self.index = self.index - 1
        return self.data[self.index]    

def reverse(data):
    for index in range(len(data)-1, -1, -1):
        yield data[index] #生成器

# 模块

模块就是一个python脚本文件，脚本名就是模块名，脚本需要以.py作为后缀名。在脚本内可以通过__name__变量读取到当前模块名。

# fibo.py 
def fib(n):    # write Fibonacci series up to n
    a, b = 0, 1
    while b < n:
        print(b, end=' ')
        a, b = b, a+b
    print()

def fib2(n):   # return Fibonacci series up to n
    result = []
    a, b = 0, 1
    while b < n:
        result.append(b)
        a, b = b, a+b
    return result

>>> import fibo

>>> fibo.__name__
'fibo'
>>> fib = fibo.fib

模块可以包含一些执行语句，这些语句是用于初始化模块的，只有在模块第一次加载的时候执行。

每个模块都有自己的一张私有变量表，模块内的变量可以安全的当作全局变量在模块各函数使用，可以直接使用modname.itemname引用模块变量。

# 包

包是用来集中管理模块的，如果一个作者有多个模块，可以把模块配置在一个包路径下，引用模块通过包名.模块名的方式引用

包目录下必须有__init__.py文件，这个是标识当前目录为一个包，另外此文件内可以添加一些执行包初始化的代码

在__init__.py定义__all__变量的值要以指定当以from xx import *时应该导入的模块列表__all__ = ["echo", "surround", "reverse"]

读写文件

open(filename, mode) 打开文件读写， mode包括r/w/r+/rb/wb/r+b

f.read([size]) 读取内容，不填size默认读取所有内容并返回

list(f) or f.readlines() 读取所有行到列表

f.write(string) 写内容到文件，返回写入量

tell() 当前读取、写入位置

f.seek(offset, from_what) 设置当前读、写位置 from_what为0则相对文件开始， 1为相对当前位置， 2为相对文件最后

class Error(Exception):
    """Base class for exceptions in this module."""
    pass

class InputError(Error):
    """Exception raised for errors in the input.

    Attributes:
        expression -- input expression in which the error occurred
        message -- explanation of the error
    """

    def __init__(self, expression, message):
        self.expression = expression
        self.message = message

class TransitionError(Error):
    """Raised when an operation attempts a state transition that's not
    allowed.

    Attributes:
        previous -- state at beginning of transition
        next -- attempted new state
        message -- explanation of why the specific transition is not allowed
    """

    def __init__(self, previous, next, message):
        self.previous = previous
        self.next = next
        self.message = message

多线程模块

import threading, zipfile

class AsyncZip(threading.Thread):
    def __init__(self, infile, outfile):
        threading.Thread.__init__(self)
        self.infile = infile
        self.outfile = outfile

    def run(self):
        f = zipfile.ZipFile(self.outfile, 'w', zipfile.ZIP_DEFLATED)
        f.write(self.infile)
        f.close()
        print('Finished background zip of:', self.infile)

background = AsyncZip('mydata.txt', 'myarchive.zip')
background.start()
print('The main program continues to run in foreground.')
background.join()    # Wait for the background task to finish
print('Main program waited until background was done.')
